'use strict';

var Record = require('./record');

/*
 * @namespace
 */
var create = {};

/**
 * Creates a new record without a model
 * @param {string} model - The name of the endpoint the record belongs to
 * @param {object} content - The content of the record
 * @function Blueprint.Data.CreateRecord
 * @example
 * var dog = Blueprint.Data.CreateRecord('pets', {
 *   'name': 'Wiley',
 *   'species': 'Dog',
 * });
 * @returns Record
 */
create.CreateRecord = function(model, content) {
  return new Record(model, {
    content: content
  });
};

module.exports = create.CreateRecord;
